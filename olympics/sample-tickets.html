var pop_first_run = false;
function popEvents () {
var cat_id = $forms.find('select[name="sport"]').val();
if (cat_id) {
var $event = $forms.find('select[name="event"]');
$event.attr('disabled',false);
$event.children().remove().end().append('<option value="">All events</option>');
for (var sid in subcategories[cat_id]) {
if(!pop_first_run && sid == util.getUrlParam('event')) {
$event.append($("<option></option").attr("value",sid).attr("selected","selected").text(subcategories[cat_id][sid]));
pop_first_run = true;
} else {
$event.append($("<option></option").attr("value",sid).text(subcategories[cat_id][sid]));
}
}
}
}

$("#clear_form_elements").click(function (e) {
$forms.find(':input').each(function() {
switch(this.type) {
case 'password':
case 'select-multiple':
case 'select-one':
case 'text':
case 'textarea':
$(this).val('');
break;
case 'checkbox':
this.checked = true;
break;
case 'radio':
this.checked = false;
}
});

});
